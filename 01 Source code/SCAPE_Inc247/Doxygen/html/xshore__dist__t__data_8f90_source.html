62  ! Cache of xshore
63  ! Originally (JT241007), three dimensions shapeFunctionID, half_tide & index within xshore.
64  ! which is very wasteful each cached xshore stored in a space big enough for the largest
65  ! as xshore is determined the number of points in the shape function + 2* half_tide this means
66  ! each record was assigned (np_sf_drift+2*cacheCeil) space i.e. 10041 when the smallest needs only 42!
67  ! New approach (JT221107) stores each element in the minimum space needed.
68  ! This array will be of length (np_sf_erode+cacheCeil)*(cacheCeil-1) + ((np_sf_drift+cacheCeil)*(cacheCeil-1))
69  ! erosion storage as (np_sf_erode+half_tide)*(half_tide-1)
70  ! drift storage will be driftOffset+((np_sf_drift+half_tide)*(half_tide-1))
71  real(kind=double), allocatable, dimension(:) :: xshorecache
72 
73  ! Index offset after which the drift data can be foumd in the cache (JT221107)
74  ! i.e. (np_sf_erode+cacheCeil)*(cacheCeil-1)

77  ! Number of points in each shape function saved, to allow better cache storage (JT221107)

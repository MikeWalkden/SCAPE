416 !> Initialises cliffy values from the base geology and layer profiles.
417 subroutine get_y_values_from_profiles(nYsections, nce, cliffy)
418  use global_data, only: base_profile, base_profile_end, layer_count, layer_profile, profile_y, baseline_od
419  use setup_module, only: nlayers, dy, dz
420  use local_data, only: logstring
421  use utilities_module, only: scape_log
422  implicit none
423 
424  integer, intent(in) :: nYsections !< number of beach sections
425  integer, intent(in) :: nce !< number of cliff elements
426  real(kind=double), intent(inout), dimension(nce, nYsections) :: cliffy !< discretised cliff y value
427 
428  integer :: section !< Beach section index
429  integer :: yIndex !< Horizontal index
430  real(kind=double) :: leftY ! Y value at left of cell
431  real(kind=double) :: rightY ! Y value at right of cell
432  real(kind=double) :: leftZ ! Z value at left of cell at top of layer stack
433  real(kind=double) :: rightZ ! Z value at right of cell at top of layer stack
434  integer :: stackIndex ! Index of layer under consideration
435  real(kind=double) :: last_base_profile_point_Y ! Y value of the last profile point for the beach section
436  real(kind=double) :: change
437  real(kind=double) :: leftZbase
438  real(kind=double) :: zAbove
439  real(kind=double) :: zBelow
440  real(kind=double), parameter :: toler = 0.000001d0
441  integer :: leftZindex
442 
443  cliffy = 0.0d0
444 
445  do section = 1, nysections
446  yindex = 0
447  last_base_profile_point_y = profile_y(base_profile_end(section))
448 
449  do
450  lefty = yindex * dy
451  righty = lefty + dy
452 
453  if (lefty > last_base_profile_point_y .or. righty > last_base_profile_point_y) then
454  exit ! Profile done
455  end if
456 
457  leftz = base_profile_value(yindex, section)
458  rightz = base_profile_value(yindex + 1, section)
459 
460  do stackindex = 1, layer_count(section)
461  leftz = leftz + layer_profile_value(yindex, stackindex, section)
462  rightz = rightz + layer_profile_value(yindex + 1, stackindex, section)
463  end do
464 
465  change = rightz - leftz
466 
467  if (abs(change) > toler) then
468  leftzbase = leftz - baseline_od
469  leftzindex = int((leftzbase + toler) / dz)
470 
471  if (leftzindex < 1) then
472  exit ! Gone below model baseline
473  end if
474 
475  if (change > 0.0d0) then
476  ! Consolidated profile slopes up
477  leftzindex = leftzindex + 1
478  zabove = leftzindex * dz + baseline_od
479 
480  if (zabove < leftz) then
481  if ((leftz - zabove) < toler) then
482  zabove = leftz
483  else
484  write(6, *)'*** Error : zAbove out of range ', zabove, lefty, leftz, righty, rightz, section
485  stop
486  end if
487  end if
488 
489  do while (zabove <= rightz)
490  cliffy(leftzindex, section) = interpolate_y_value(zabove, lefty, leftz, righty, rightz)
491  leftzindex = leftzindex + 1
492  zabove = zabove + dz
493  end do
494  else
495  ! Consolidated profile slopes down
496  zbelow = leftzindex * dz + baseline_od
497 
498  if (zbelow > leftz) then
499  if ((zbelow - leftz) < toler) then
500  zbelow = leftz
501  else
502  write(6, *)'*** Error : zBelow out of range ', zbelow, lefty, leftz, righty, rightz, section
503  stop
504  end if
505  end if
506 
507  do while (zbelow >= rightz)
508  cliffy(leftzindex, section) = interpolate_y_value(zbelow, lefty, leftz, righty, rightz)
509  leftzindex = leftzindex - 1
510  zbelow = zbelow - dz
511  end do
512  end if
513  end if
514 
515  yindex = yindex + 1 ! Move to next cell further offshore
516  end do
517  end do ! section = 1, nYsections
518 
519 end subroutine get_y_values_from_profiles

522 !> Cut the base geology and layer profiles to match current cliffy values.
523 !>
524 !> Only modifies the profiles in the range Y = 0 to YmaxH
525 subroutine cut_layer_profiles(nYsections, nce, cliffy)
526  use global_data, only: base_profile, layer_count, layer_profile, msl_m, msl_od
527  use setup_module, only: nlayers, dy, dz
528  implicit none
529 
530  integer, intent(in) :: nYsections !< number of beach sections
531  integer, intent(in) :: nce !< number of cliff elements
532  real(kind=double), intent(inout), dimension(nce, nYsections) :: cliffy !< discretised cliff y value
533 
534  real(kind=double), dimension(0: nHcells) :: profile ! Z value of profile from current cliffy values
535  integer :: section !< Beach section index
536  integer :: yIndex !< Horizontal index
537  integer :: zIndex !< Vertical index
538  real(kind=double) :: zValue ! Z value up the layer stack
539  real(kind=double) :: midZ ! Z value midway up cell
540  real(kind=double) :: previousY ! Y value at previous Y values point
541  real(kind=double) :: previousZ ! Z valueat previous Y values point
542  integer :: stackIndex !< Index of layer under consideration
543  logical :: firstValueSet
544 
545  do section = 1, nysections
546  yindex = 0
547  firstvalueset = .false.
548 
549  ! Construct a profile to match the current Y values
550  do zindex = nce, 1, -1
551  midz = dz * (real(zIndex - 1, kind=double) + 0.5D0) - msl_m + msl_od
552 
553  if (.not. firstvalueset .and. cliffy(zindex, section) > 0.0d0) then
554  profile(yindex) = midz
555  firstvalueset = .true.
556  yindex = yindex + 1
557  previousy = 0.0d0
558  previousz = midz
559  end if
560 
561  if (firstvalueset) then
562  do while ((yindex * dy) <= cliffy(zindex, section) .and. yindex <= nhcells)
563  profile(yindex) = interpolate_z_value(yindex * dy, previousy, previousz, cliffy(zindex, section), midz)
564  yindex = yindex + 1
565  end do
566  end if
567 
568  previousy = cliffy(zindex, section)
569  previousz = midz
570  end do ! zIndex = nce, 1, -1
571 
572  if (yindex < nhcells + 1) then ! extend the profile to YmaxH if needed
573  profile(yindex: nhcells) = previousz
574  end if
575 
576  do yindex = 0, nhcells
577  if (base_profile(yindex, section) > profile(yindex)) then
578  base_profile(yindex, section) = profile(yindex)
579  end if
580 
581  zvalue = base_profile(yindex, section)
582 
583  do stackindex = 1, layer_count(section)
584  if (zvalue + layer_profile(yindex, stackindex, section) > profile(yindex)) then
585  layer_profile(yindex, stackindex, section) = profile(yindex) - zvalue
586  end if
587 
588  zvalue = zvalue + base_profile(yindex, section)
589  end do ! stackIndex = 1, layer_count(section)
590  end do ! yIndex = 0, nHcells
591  end do ! section = 1, nYsections
592 
593 end subroutine cut_layer_profiles

634 !>
635 !> Calculates the number of cells required to achieve the required volume as a layer of the required thickness.
636 !> Centres the cells near the required yPosition. Adds a two cell wide taper to the layer at each end.
637 !>
638 !> The requested volume is always achieved exactly. The position and thickness arguments are treated as targets.
639 subroutine add_material_to_layer(nLayer, section, volume, thickness, yPosition, success_code)
640  use global_data, only: layer_count, layer_profile
641  use local_data, only: logstring
642  use setup_module, only: dx, dy
643  use exceptions
644  implicit none
645 
646  integer, intent(in) :: nLayer !< Layer index
647  integer, intent(in) :: section !< Beach section index
648  real(kind=double), intent(in) :: volume !< Volume of material to add to layer
649  real(kind=double), intent(in) :: thickness !< Target maximum thickness of added material
650  real(kind=double), intent(in) :: yPosition !< Target y position at middle of added material
651  integer, intent(out) :: success_code !< 0: ok, -ive fatal error, +ive warning
652 
653  integer :: yIndex !< Y value index
654  integer :: startIndex !< Y value index of first cell taking material
655  integer :: endIndex !< Y value index of last cell taking material
656  integer :: nCells !< Number of cells required to take the material, not including the startIndex and endIndex cells
657  real(kind=double) :: area !< Area to be added to the profile for the layer
658  real(kind=double) :: actualThickness !< Actual layer thickness, calculated to get the area right after the number of cells has been determined
659 
660  if (nlayer < 1 .or. nlayer > layer_count(section)) then
661  write(logstring,*) 'Layer index ', nlayer, ' out of range at beach section ', section
662  call raise_exception(logstring)
663  success_code = -1
664  return
665  end if
666 
667  if (volume <= 0.0d0) then ! No work to do
668  return
669  end if
670 
671  area = volume / dx
672  ncells = int(area / (thickness * dy))
673 
674  if (ncells == 0) ncells = 1 ! Must have at least one cell
675 
676  yindex = nint(yposition / dy)
677  startindex = yindex - (ncells / 2)
678 
679  if (mod(ncells, 2) == 1) then
680  startindex = startindex - 1
681  end if
682 
683  endIndex = yIndex + (nCells / 2) + 1
684 
685  do while (startindex < 0) ! Make sure startIndex is within range
686  startindex = startindex + 1
687 
688  if (endindex < nhcells) then
689  endIndex = endIndex + 1
690  end if
691  end do
692 
693  do while (endindex > nhcells) ! Make sure endIndex is within range
694  endIndex = endIndex - 1
695 
696  if (startindex > 0) then
697  startindex = startindex - 1
698  end if
699  end do
700 
701  ncells = endindex - startindex - 1 ! Equivalent number of full height cells
702  actualthickness = area / (ncells * dy)
703  ! material added to startIndex cell tapers from 0 to half actuallThickness
704  layer_profile(startindex + 1, nlayer, section) = layer_profile(startindex + 1, nlayer, section) + 0.5 * actualthickness
705  ! material added to endIndex cell tapers from half actuallThickness to 0
706  layer_profile(endindex, nlayer, section) = layer_profile(endindex, nlayer, section) + 0.5 * actualthickness
707 
708  do yindex = startindex + 2, endindex - 1
709  ! material added to cells in the middle has depth actuallThickness
710  layer_profile(yindex, nlayer, section) = layer_profile(yindex, nlayer, section) + actualthickness
711  end do

752 !> Returns an int indicating whether the HA grid cell of interest is empty, or which of the base geology layer
753 !> or one of the layers above covers most of the sell area.
754 function get_cell_dominant_layer(zValue, yIndex, section) result(layerSelected)
755  use global_data, only: base_profile, layer_count, layer_profile, baseline_od
756  use setup_module, only: nlayers
757  implicit none

760  integer, intent(in) :: yIndex !< Horizontal index
761  integer, intent(in) :: section !< Beach section index

763  !< Greater than 0 indicates which of the upper layers covers most of the cell area.

765  real(kind=double) :: lowZ ! Z value at bottom of cell of interest
766  real(kind=double) :: highZ ! Z value at top of cell of interest
767  real(kind=double) :: leftY ! Y value at left of cell of interest
768  real(kind=double) :: rightY ! Y value at right of cell of interest
769  real(kind=double) :: cellArea ! Area of a cell

771  real(kind=double) :: areaBelow ! Area of the cell of interest below the layer currently under consideration
772  real(kind=double) :: areaThisLayer ! Area of the cell of interest occupied by the layer under consideration
773  real(kind=double) :: leftZ ! Z value at top of layer under consideration at left of cell of interest
774  real(kind=double) :: rightZ ! Z value at top of layer under consideration at right of cell of interest
775  real(kind=double) :: yInterpolated
776  integer :: stackIndex !< Index of layer under consideration
777  real(kind=double) :: selectedLayerArea !< Area of the cell covered by the selected layer
778 
779  lefty = yindex * dy
780  righty = lefty + dy
781  lowz = int((zvalue - baseline_od) / dz) * dz + baseline_od
782  highz = lowz + dz
783 
784  cellarea = dy * dz
785  stackindex = 0
786  areacovered = 0.0d0
787  layerselected = -1
788  selectedlayerarea = 0.0d0
789  leftz = base_profile_value(yindex, section)
790  rightz = base_profile_value(yindex + 1, section)
791 
792  do while (areacovered < cellarea)
793  ! Calculate the area of the cell of interest below the layer currently under consideration
794  if (leftz < lowz) then
795  if (rightz < lowz) then
796  areabelow = 0.0d0
797  else if (rightz >= lowz .and. rightz <= highz) then
798  areabelow = 0.5d0 * (righty - interpolate_y_value(lowz, lefty, leftz, righty, rightz)) * (rightz - lowz)
799  else ! rightZ > highZ
800  areabelow = 0.5d0 * (interpolate_y_value(lowz, lefty, leftz, righty, rightz) + &
801  interpolate_y_value(highz, lefty, leftz, righty, rightz)) * dz
802  end if
803  else if (leftz >= lowz .and. leftz <= highz) then
804  if (rightz < lowz) then
805  areabelow = 0.5d0 * interpolate_y_value(lowz, lefty, leftz, righty, rightz) * (leftz - lowz)
806  else if (rightz >= lowz .and. rightz <= highz) then
807  areabelow = 0.5d0 * (leftz - lowz + rightz - lowz) * dy
808  else ! rightZ > highZ
809  yinterpolated = interpolate_y_value(highz, lefty, leftz, righty, rightz)
810  areabelow = 0.5d0 * (leftz - lowz + dz) * (yinterpolated - lefty) + dz * (righty - yinterpolated)
811  end if
812  else ! leftZ > highZ
813  if (rightz < lowz) then
814  areabelow = 0.5d0 * (interpolate_y_value(lowz, lefty, leftz, righty, rightz) + &
815  interpolate_y_value(highz, lefty, leftz, righty, rightz)) * dz
816  else if (rightz >= lowz .and. rightz <= highz) then
817  yinterpolated = interpolate_y_value(highz, lefty, leftz, righty, rightz)
818  areabelow = 0.5d0 * (rightz - lowz + dz) * (righty - yinterpolated) + dz * (yinterpolated - lefty)
819  else ! rightZ > highZ
820  areabelow = dy * dz
821  end if
822  end if
823 
824  areathislayer = areabelow - areacovered
825 
826  if (areathislayer > selectedlayerarea) then
827  selectedlayerarea = areathislayer
828  layerselected = stackindex
829 
830  if (selectedlayerarea >= 0.5d0 * cellarea) exit
831  end if
832 
833  areacovered = areabelow
834 
835  stackindex = stackindex + 1
836 
837  if (stackindex > layer_count(section) .or. yindex + 1 > nhcells) exit
838 
839  leftz = leftz + layer_profile_value(yindex, stackindex, section)
840  rightz = rightz + layer_profile_value(yindex + 1, stackindex, section)
841  end do

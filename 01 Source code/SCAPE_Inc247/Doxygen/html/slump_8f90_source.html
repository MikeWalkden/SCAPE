72 
73  cliffy(1, section) = cliffy(2, section) + 1
74 
75  ! Slump the cliff toe (upper platform)
76  do j=1,nce-1
77  overhang = cliffy(j+1, section) - cliffy(j, section)
78 
79  if(overhang > 0)then
80  cliffy(j+1, section) = cliffy(j, section)
81  tvolume( section) = tvolume( section) + overhang * dx * dz
82  end if
83 
84  end do
85 
86  ! Slump the main body of the cliff
87  tvolume(section) = tvolume(section) + ((clifftoe_pos(section) - cliffy(nce, section)) &
88  * dx * cliff_heights(section))
89 
90  clifftoe_pos(section) = cliffy(nce, section)
91 
92  return

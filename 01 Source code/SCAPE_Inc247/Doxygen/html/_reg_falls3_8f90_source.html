5 !****************************************************************************
6 !> This is a probabalistic model of landsliding that is separate from SCAPE,
7 !> runs after it. It takes the projections of cliff toe recession from SCAPE
8 !> and uses them to predict cliff top location. It does this many times,
9 !> randomly selecting sequences of landslide period and size. In this way
10 !> an ensemble set of cliff top recession estimates is created.
11 !> See Dawson, R., Dickson, M., Nichols, R., Hall, J., Walkden, M., Stansby, P.,
12 !> Mokrech, M., Richards, J., Zhou, J., Milligan, J., Jordan, A., Pearson, S.,
13 !> Rees, S., Bates, P., Koukoulas, S., Watkinson, A., (2009)
14 !> Integrated analysis of risks of coastal flooding and cliff erosion under
15 !> scenarios of long term change. Climatic Change, Climatic Change (2009)
16 !> 95:249-288. DOI 10.1007/s10584-008-9532-8. Springer.

74  character (*), intent(in) :: filename !< name of cliff data file

108  if (success_code < 0) return
109

111  if (success_code < 0) return

120 
121  call random_seed(put=seed)

123 ! Read in data describing the cliff
124 ! cliffdata.txt (8 by nCliffSections) is [cliff height; current angle mean; current angle sd; pre-fall angle mean;
125 ! sd; post-fall angle mean; sd; predsect]
126 
127  do
128  ctop = 0
129 
130  read(cliffdataunit, *, iostat=iostat) cliffdata
131  if (iostat == -1) exit ! end of file
132  success_code = success_code_from_iostat(cliffdataunit, iostat, 'Reading cliffData')
133  if (success_code < 0) exit
134 
135  cht = cliffdata(1) ! cliff height
136  meaninia = cliffdata(2) ! Initial mean angle
137  sdinia = sqrt(cliffdata(3)) ! Initial angle stdeviation
138  meanfalla = cliffdata(4) ! Mean falling angle
139  sdfalla = sqrt(cliffdata(5)) ! Stdeviation of falling angle
140  meanstaba = cliffdata(6) ! Mean stable angle, post fall
141  sdstaba = sqrt(cliffdata(7)) ! Stdeviation stable angle
142  cpsect = cliffdata(8) ! SCAPE section number
143 
144  toeret = recession(cpsect, :) !Toe retreat
145 
146  !simulate cliff angles before and after fall
147  call normal(rand_ns, ncliffsimulations)
148 
149  iniasim = (rand_ns * sdinia) + meaninia
150 
151  do i = 1, ncliffsimulations
152  ctop(:, i) = cht(i) / tan(iniasim(i) * 3.142 / 180) ! Distance behind the local cliff toe
153  enddo
154 
155  havefallinfo = .false.
156 
157  do i = 1,nyears
158  !for each simulation calculate new angle to cliff top based on how much the toe has retreated
159  angle = atan(cht / (ctop(i, :) + toeret(i))) * 180 / 3.142 ! toe retreat is negative
160 
161  ! If a failure occurs then recalculate the cliff top position
162  do j = 1, ncliffsimulations
163  if (.not. havefallinfo) then
164  call normal(rand_ns, ncliffsimulations)
165  fallasim = (rand_ns * sdfalla) + meanfalla
166 
167  ! To simulate cliff stable angle stabasim
168  call normal(rand_ns, ncliffsimulations)
169  stabasim = (rand_ns * sdstaba) + meanstaba ! cliff stable angle
170 
171  havefallinfo = .true.
172  end if
173 
174  if ((angle(j) .gt. fallasim(j) .and. stabasim(j) < angle(j)) .or. angle(j) < 0) then
175  ! Finds the difference in horizontal position associated with the two angles
176  ctop(i,j) = ctop(i,j) + (cht(j) / (tan(stabasim(j) * 3.142 / 180)) - cht(j) / tan(angle(j) * 3.142 / 180))
177 
178  do m = i + 1, nyears
179  ctop(m, j) = ctop(i, j)
180  enddo
181 
182  !ctop(i:nyears,j) = ctop(i,j) + [cht(j) / (tan(stabasim(j)*3.142/180)) - cht(j)/tan(angle(j)*3.142/180)]
183 
184  havefallinfo = .false.
185  endif
186  enddo
187 
188  enddo
189 
190 
191  do i = 1,nyears ! years
192  do j = 1, ncliffsimulations ! simulations
193  if (i == 1) then ! year 1 so save initial data
194  write(saveclifftopunit,"(I3)") cpsect ! seciton
195  write(saveclifftopunit,"(I3)") j ! simulation
196  write(saveclifftopunit,"(I3)") i ! year
197  write(saveclifftopunit,"(F5.1)") ctop(i,j) ! offset
198  elseif (ctop(i,j) /= ctop(i-1,j)) then
199  write(saveclifftopunit,"(I3)") cpsect
200  write(saveclifftopunit,"(I3)") j ! simulation
201  write(saveclifftopunit,"(I3)") i ! year
202  write(saveclifftopunit,"(F5.1)") ctop(i,j) ! offset
203  endif
204  enddo
205  enddo
206 
207  enddo
208 
209  close(cliffdataunit)
210 
211  close(saveclifftopunit)

105  if (iprofile == -1) then ! loop over all nYsections
106  first = 1
107  last = nysections
108  else

111  end if

113  do section = first, last
114 
115  if (beach_type == beach_vellinga) then
116 
117  ! Find where the beach overlies the platform
118  do index = vell_top_e + 1, nce
119  vellinga(index) = 0
120  enddo
121 
122  do index = 1, vell_top_e
123  vellinga(index) = seawall_os(section) + berm(section) + vxs(index)
124  enddo
125 
126  ! SCAPE describes the beach volume twice, once with precision
127  ! in the sediment transport / sediment budget calculations using the name:
128  ! bvolume(i)
129  ! and a second approximate value:
130  ! bVolumeApprox
131  ! which is the volume calculated by summing the difference between the
132  ! discretised VELLINGA PROFILE and the discretised platform (cliffy(,)).
133  ! bVolumeApprox is changed by moving the discretised beach
134  ! horizontally, until it is similar to bvolume(i).
135 
136  bvolumeapprox = 0.0d0
137 
138  do index = 1, nce
139  if ((vellinga(index) - cliffy(index, section)) .gt. 0)then
140  bvolumeapprox = bvolumeapprox + (vellinga(index) - cliffy(index, section)) * dz * dx
141  endif
142  enddo
143 
144  else ! regular beach, upper beach or lower beach
145  exposed_surface(bottom(section): top(section)) = &
146  max(cliffy(bottom(section): top(section), section), surface(bottom(section): top(section), section))
147 
148  beachdepth = 0
149 
150  ! Find where the beach overlies the platform
151  do index = bottom(section), top(section)
152  if (beachy(index, section) > exposed_surface(index)) then
153  beachdepth(index) = beachy(index, section) - exposed_surface(index)
154  end if
155  end do
156 
157  ! SCAPE describes the beach volume twice, once with precision
158  ! in the sediment transport / sediment budget calculations using the name:
159  ! bvolume(i)
160  ! and a second approximate value:
161  ! bVolumeApprox
162  ! which is the volume calculated by summing the difference between the
163  ! discretised beach (beachy(,)) and the discretised platform (cliffy(,)).
164  ! bVolumeApprox is changed by moving the discretised beach
165  ! horizontally, until it is similar to bvolume(i).
166 
167  bvolumeapprox = sum(beachdepth(bottom(section): top(section))) * dz * dx
168 
169  if (beach_type == beach_regular .and. bermslope > 0.0d0) then
170  if (beachy(beachelev(section) - 1, section) > cliffy(beachelev(section) - 1, section)) then
171  bvolumeapprox = bvolumeapprox + ((0.5d0 * dx * (beachy(beachelev(section) - 1, section) - &
172  cliffy(beachelev(section) - 1, section))**2) / bermslope)
173  ! This has been added 80(13) to account for the volume of material assumed by
174  ! PLATFORM_EROSION to be perched on top of any berm
175  ! This is only a rough estimate, it should be improved laer, in particular to
176  ! take account of the presence of any talus.
177  ! NB the same calculation is made again below
178  end if
179  endif
180  endif
181 
182  bermflag = 0 ! flag for direction of change
183  bermflag1 = 0 ! flag for change in change direction
184  bermstep = initialbermstep
185 
186  do while(bvolumeapprox > 1.05 * bvolume(section) .or. bvolumeapprox < 0.95 * bvolume(section))
187  if(bvolumeapprox < bvolume(section))then ! The estimate of volume is too small
188  if(bermflag == -1)then ! Last time the app beach was too large
189  call halve_berm_step(bermstep) ! Modify step size
190  bermflag1 = 1 ! there has been a change in direction
191  else if(bermflag == 1 .and. bermflag1 == 0)then
192  call double_berm_step(bermstep) ! Modify step size
193  end if
194 
195  ! Move the approximate beach out
196  if (beach_type == beach_upper .or. beach_type == beach_lower) then
197  beachos(section) = beachos(section) + bermstep
198  else
199  berm(section) = berm(section) + bermstep
200  end if
201 
202  bermflag = 1
203  else if(bvolumeapprox > bvolume(section))then ! The approximate beach is too large
204  if(bermflag == 1)then ! last time the app beach was too small
205  call halve_berm_step(bermstep) ! Modify step size
206  bermflag1 = 1 ! there has been a change in direction,
207  else if(bermflag == -1 .and. bermflag1 == 0)then
208  call double_berm_step(bermstep) ! Modify step size
209  end if
210 
211  ! Move the approximate beach in
212  if (beach_type == beach_upper .or. beach_type == beach_lower) then
213  beachos(section) = beachos(section) - bermstep
214  else
215  berm(section) = berm(section) - bermstep
216  end if
217 
218  bermflag = -1
219  end if
220 
221  if (beach_type == beach_vellinga) then
222 
223  ! Find where the beach overlies the platform
224  do index = vell_top_e + 1, nce
225  vellinga(index) = 0
226  enddo
227 
228  do index = 1, vell_top_e
229  vellinga(index) = seawall_os(section) + berm(section) + vxs(index)
230  enddo
231 
232  bvolumeapprox = 0.0d0
233 
234  do index = 1, nce
235  if ((vellinga(index) - cliffy(index, section)) .gt. 0)then
236  bvolumeapprox = bvolumeapprox + (vellinga(index) - cliffy(index, section)) * dz * dx
237  endif
238  enddo
239 
240  else ! regular beach, upper beach or lower beach
241  ! Now the approximate beach has moved, check the approximate beach volume
242  ! First 'redraw' the approximate beach
243  if (beach_type == beach_regular) then
244  beachy(top(section) + 1: bottom(section) + 1: -1, section) = beachos(section) + berm(section) + bruun
245  else
246  beachy(top(section): bottom(section): -1, section) = beachos(section) + bruun
247  end if
248 
249  ! earlier verrsion in WkBk 9(97)
250 
251  beachdepth = 0
252 
253  ! Find where the beach overlies the platform
254  do index = bottom(section), top(section)
255  if (beachy(index, section) > exposed_surface(index)) then
256  beachdepth(index) = beachy(index, section) - exposed_surface(index)
257  end if
258  end do
259 
260  ! bVolumeApprox is the aparent beach volume (see above), it has to be
261  ! adjusted to (approximately) match the actual beach volume
262  bvolumeapprox = sum(beachdepth(bottom(section): top(section))) * dz * dx
263 
264  if (beach_type == beach_regular .and. bermslope > 0.0d0) then
265  if (beachy(beachelev(section)-1,section) > cliffy(beachelev(section)-1, section))then
266  bvolumeapprox = bvolumeapprox + ((0.5d0 * dx*(beachy(beachelev(section) - 1,section) - &
267  cliffy(beachelev(section) - 1, section))**2) / bermslope)
268  ! The extra volume has been added 80(13) to account for the volume of material assumed by
269  ! PLATFORM_EROSION to be perched on top of any berm
270  ! This is only a rough estimate, it should be imporved laer, in particular to
271  ! take account of the presence of any talus
272  ! NB the same calculation is made again above
273  end if
274  end if
275  end if
276 
277  ! Rounding check to prevent over calculation
278  if(nint(bvolumeapprox) == nint(bvolume(section)))then
279  bvolumeapprox = bvolume(section)
280  end if
281  end do
282  end do
283 
284  return

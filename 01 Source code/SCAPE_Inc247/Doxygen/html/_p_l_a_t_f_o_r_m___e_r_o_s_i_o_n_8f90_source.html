6 !
7 !> This SCAPE module describes the erosion of sections of platform, and lower cliff.
8 !>
9 !> Representation of 'storm' beach behaviour.
10 !> This was introduced to represent beach behaviour during a storm. It was
11 !> decided to do this because:
12 !> (a) the regional beach is unresponsive to individual waves, so is more
13 !> stable than reality in that it is less likley to empty and reveal the
14 !> platform to wave attack,
15 !> (b) during storms material is moved offshore making the platform more
16 !> vulnerable,
17 !> (c) the storm beach profile is lower and flatter, again leaving the
18 !> platform more vulnerable.
19 !>
20 !> The algorithms here try to represent these things by dividing the beach
21 !> into two parts when a storm occurs. Each part (upper and lower) has its own
22 !> volume and its own discretised surface. The upper beach is steeper than the
23 !> usual beach, the lower beach is flatter. Material can move from the upper to
24 !> the lower during a storm. The rates of movement are determined through a user defined look-up table.
25 !> Once the storm is over the beach returns to its 'normal' form, and its colume recovers using a behavioural rule.

528  end do

530  temp5 = bdepth / (beachdisturbanceratio * h(section))
531  where (temp5 > 1.0d0)
532  temp5 = 1.0d0
533  end where
534  battenuation = 1.0d0 - temp5 ! This is the attenuation of the erosive capability of the
535  ! beach. 1 indicates no beach protection.
536 
537  if (tempbeachy(beachelev(section) + 1, section) > 0)then
538  ! reset the top of the beach to a flat slope, return it to 'normal'
539  ! following the modification of tempBeachy to represent the berm slope
540  do j = beachelev(section) + 1, nce
541  tempbeachy(j, section) = 0
542  enddo
543  endif
544

570 subroutine crossbeach_exchange(i, nYsections, nce, ubTop, ubBottom, lbTop, lbBottom, &
571  upbeachheighte, lowbeachheighte, &
572  stormflag, crossbeach_st, &
573  upbvolume, upbruun, upbeach_os, upbeachy, &

576  period, tstep_secs)
577

579  use setup_module
580  implicit none
581 
582  integer, intent(in) :: i !< beach section index
583  integer, intent(in) :: nYsections !< number of beach sections
584  integer, intent(in) :: nce !< number of platform elements
585  integer, intent(in), dimension(nYsections) :: ubTop !< indices of upper beach top elements
586  integer, intent(in), dimension(nYsections) :: ubBottom !< indices of upper beach bottom elements
587  integer, intent(in), dimension(nYsections) :: lbTop !< indices of lower beach top elements
588  integer, intent(in), dimension(nYsections) :: lbBottom !< indices of lower beach bottom elements
589  integer, intent(in) :: upBeachHeightE !< upper beach height in number of elements
590  integer, intent(in) :: lowBeachHeightE !< lower beach height in number of elements
591  integer, intent(inout), dimension(nYsections) :: stormflag !< 0 indicates first timestep of sequence of storm waves
592  real(kind=double), intent(inout), dimension(nYsections) :: crossbeach_st !< Cross beach sediment transport
593  real(kind=double), intent(inout), dimension(nYsections) :: upbvolume !< upper beach volume
594  real(kind=double), intent(in), dimension(upBeachHeightE) :: upbruun !< upper beach 'Bruun' shape
595  real(kind=double), intent(inout), dimension(nYsections) :: upbeach_os !< Upper beach offset at each beach section
596  real(kind=double), intent(inout), dimension(nce,nYsections) :: upbeachy !< distance of surface of upper beach from baseline, active during storm conditions
597  real(kind=double), intent(inout), dimension(nYsections) :: lowbvolume !< lower beach volume
598  real(kind=double), intent(in), dimension(lowBeachHeightE) :: lowbruun !< lower beach 'Bruun' shape
599  real(kind=double), intent(inout), dimension(nYsections) :: lowbeach_os !< Lower beach offset at each beach section
600  real(kind=double), intent(inout), dimension(nce, nYsections) :: lowbeachy !< distance of surface of lower beach from baseline, active during storm conditions
601  real(kind=double), intent(in), dimension(nce, nYsections) :: cliffy !< Discretised cliff and platform y value
602  real(kind=double), intent(in), dimension(nce, nYsections) :: talusy !< Talus y value (horizontal distance of talus surface from baseline)
603  real(kind=double), intent(inout), dimension(nce) :: beachdepth !< Depth of the beach
604  real(kind=double), intent(in), dimension(nYsections) :: bvolume !< Beach volume
605  real(kind=double), intent(in), dimension(nYsections) :: h !< Wave height
606  real(kind=double), intent(in), dimension(nYsections) :: offshore_st !< Offshore sediment transport
607  real(kind=double), intent(in) :: period !< Wave period
608  real(kind=double), intent(in) :: tstep_secs !< Time step in seconds
609 
610  integer :: index
611  real(kind=double) :: cbstrLookup
612 
613  if(stormflag(i) == 0)then ! this is the first timestep of
614  ! this sequence of storm waves
615  stormflag(i) = 1
616  ! Obtain values for the upper and lower beach volumes
617  upbvolume(i) = bvolume(i)/2 !Upper beach volume
618  lowbvolume(i) = upbvolume(i) !lower beach volume
619  else ! This is not the first timestep, and we have to
620  !account for the loss of material to the offshore, this
621  ! will have already been calculated in SEDIMENT_TRANSPORT, but the beach
622  ! volumes in the storm sequence
623  ! are disconnected from the rest of the model
624  lowbvolume(i) = lowbvolume(i) + offshore_st(i)
625  if(lowbvolume(i) < 0)then
626  lowbvolume(i) = 0
627  endif
628  endif
629 
630  cbstrlookup = crossbeach_str_lookup(h(i), period)
631  crossbeach_st(i) = cbstrlookup * dx * tstep_secs / 3600 ! convert to units suitable for this timestep
632 
633  if(-1*crossbeach_st(i) > upbvolume(i))then
634  crossbeach_st(i) = -1*upbvolume(i)
635  endif
636  upbvolume(i) = upbvolume(i) + crossbeach_st(i)
637  lowbvolume(i) = lowbvolume(i) - crossbeach_st(i)
638 
639 
640 !*****************************************************
641 ! Now find the positions of the UPPER beach
642 !*****************************************************
643 
644  call beach_location(i, nysections, nce, upbeachheighte, ubtop, ubbottom, &
645  upbruun, upbeach_os, upbeachy, &
646  cliffy, beachdepth, upbvolume, &
647  talusy, beach_upper)
648 
649 !*****************************************************
650  ! Now find the positions of the LOWER beach
651 !*****************************************************
652 
653  call beach_location(i, nysections, nce, lowbeachheighte, lbtop, lbbottom, &
654  lowbruun, lowbeach_os, lowbeachy, &
655  cliffy, beachdepth, lowbvolume, &
656  upbeachy, beach_lower)
657 
658 ! End of crossbeach exchange calculations
659 
660 end subroutine crossbeach_exchange

57 !> Get sea level for given year from Sea Level file.
58 subroutine sl_for_year(year, sl, success_code)
59  integer, intent(in) :: year !< year for which sea level is required
60  real(kind=double), intent(inout) :: sl !< sea level found
61  integer, intent(out) :: success_code !< 0: ok, -ive fatal error, +ive warning
62 
63  integer :: ios
64 
65  success_code = 0
66 
67  if (slfileunit == 0) then ! first time through
68  slfileunit = get_file_unit()
69  success_code = do_open_text_input_file(slfileunit, file_path, sealevelfile)
70  if (success_code < 0) return
71  end if
72 
73  do
74  read(slfileunit, *, iostat = ios) slyear, sl
75 
76  if (ios == -1) then
77  exit
78  else
79  success_code = success_code_from_iostat(slfileunit, ios, 'Reading sea level data')
80  if (success_code < 0) return
81  end if
82 
83  if (slyear >= year) then
84  exit
85  end if
86  end do
87 
88  if (slyear /= year) then
89  write(felogstring, *) 'Could not find SL data for year ', year
90  call raise_exception(felogstring)
91  success_code = -1
92  return
93  end if
94 
95 end subroutine sl_for_year

54 !> This routine does a numerical integration, stepping the
55 !> shape function through the tidal cycle and summing it
56 !> each step to produce a distribution representing the
57 !> whole tidal cycle.
58 !>
59 !> This process has to represent the different amount of
60 !> time that the tide spends at the different levels, more
61 !> at the extremes thena in the middle.
62  subroutine xshore_dist_t(half_tide, sf, nce, dbreak, &
63  dsf, setup, heightsurge, np_sf, lowlim, uplim, &
64  cliffheights, msl_m, this_dist, bottom, top, sfid, success_code)

75  real(kind=double), dimension(50, 2), intent(in) :: sf !< The shape function (of drift or erosion,
76  !< under breaking waves with a static water level
77  integer, intent(in) :: nce !< Number of cliff elements

81  real(kind=double), intent(in) :: heightsurge !< Not used, left at zero

84  integer, intent(in) :: uplim !< The upper limit of the model (nce)

110  ! npp the number of points in the distribution, with dx = the
111  ! dx of the shape function.
112  npp = np_sf + 2 * half_tide
113 
114  !*** Allocate arrays
115 
116  !*** Allocate arrays if they not big enough (JT221107a)
117  if (half_tide > allocceil)then
118  deallocate(xshore, tide_duration, start_fnctn, &
119  end_fnctn, temp, stat=alloc_error)

121  write(6,*)'*** Unexpected deallocation error'

124

129  success_code = success_code_from_allocation(alloc_error, 'subroutine xshore_dist_t')
130  if (success_code < 0) return
131 
132  endif
133  !*** end of alloc (JT221107a)
134 
135  ! calculate the offset where the cache will be stored (JT221107)
136  if (sfid == 1)then
137  xshorecacheoffset = driftoffset+((np_sf+half_tide)*(half_tide-1))
138  else
139  xshorecacheoffset = (np_sf+half_tide)*(half_tide-1)
140  endif
141 
142  ! retrieve xshore if already cached (JT241007) mem optimised (JT221107)
143  alreadycached = half_tide < cacheceil
144  if (alreadycached) alreadycached = xshorecache_flag(sfid, half_tide)
145 
146  if (alreadycached) then
147  xshore(1:npp)=xshorecache((xshorecacheoffset+1):(xshorecacheoffset+npp))
148  else
149  ! otherwise calculate xshore (JT241007)
150 
151  ! Account for the sin variation of the tide
152  ! The tide is symmetrical so only have to deal with a part of it
153  ! Calculate the duration of the time spent at each elevation
154 
155  tide_duration = 0.0d0
156  counter1 = half_tide
157  counter2 = half_tide+1
158  tide_duration(1) = 1
159 
160  do j=1,half_tide
161  jj = j
162  counter1 = counter1+1
163  counter2 = counter2-1
164  first = (jj-1)/half_tide
165  second = jj/half_tide
166  tide_duration(counter1) = 2.0d0*(asin(second)-asin(first))
167  tide_duration(counter2) = tide_duration(counter1)
168  end do
169 
170  tide_duration = tide_duration / 6.283d0
171 
172  ! The start and end functions define start and end positions in the shape function.
173  start_fnctn(1:npp) = 1
174  do j=1,np_sf+1
175  start_fnctn(npp-np_sf-1+j) = j
176  end do
177  do j=1,npp
178  if(j>=np_sf)then
179  end_fnctn(j) = np_sf
180  else
181  end_fnctn(j) = j
182  end if
183  end do
184 
185  xshore(1:npp) = 0.0d0
186 
187  ! start of corrected version see 105 wkbk 9
188  counter1 = 0
189  do j = 1, npp - 1
190  counter1 = counter1+1
191  if(counter1 > 2*half_tide) then
192  counter1 = 2*half_tide
193  endif
194  ! check for unreal situation of no tide
195  if(counter1 == 0)counter1=1
196 
197  locatesf(1:(end_fnctn(j)-start_fnctn(j)+1))=sf(start_fnctn(j):end_fnctn(j),1)
198 
199  duration_sub(1:(end_fnctn(j)-start_fnctn(j)+1)) = tide_duration(counter1:counter1-(end_fnctn(j)-start_fnctn(j)):-1)
200 
201  do k=1,(end_fnctn(j)-start_fnctn(j)+1)
202  xshore(j) = xshore(j) + (locatesf(k)*duration_sub(k));
203  enddo
204  enddo
205  ! end of corrected version see 105 wkbk 9
206 
207  ! store xshore if under cache ceiling (JT241007) mem optimised (JT221107)
208  if (half_tide < cacheceil) then
209  xshorecache((xshorecacheoffset + 1): (xshorecacheoffset + npp)) = xshore(1:npp)
210  xshorecache_flag(sfid, half_tide) = .true.
211  endif
212 
213  ! end of caching brackets (JT241007)
214  endif
215 
216  do j = 1, npp
217  temp(j) = xshore(npp + 1 - j)
218  end do
219  xshore(1: npp) = temp(1: npp)
220 
221  this_dist=0.0d0
222 
223  dzdist = dbreak * dsf
224  distlow = msl_m + heightsurge + setup - ((half_tide + np_sf)*dzdist)
225  disthigh = msl_m + heightsurge + setup + (half_tide * dzdist)
226  top = nint(disthigh / dz)
227  bottom = nint(distlow / dz)
228 
229  if ((top < lowlim) .or. (bottom > uplim - 1)) then
230  write(logstring, fmt='(A,I4,A,I4,A,I4,A,I4,A)') &
231  ' xshore_dist_t : distribution (', bottom, ':', top, ') model (', lowlim, ':', uplim, ')'
232  call raise_exception(logstring)
233  success_code = -1
234  return
235  end if
236 
237  if (bottom < lowlim) bottom = lowlim
238  if (top > uplim-1) top = uplim - 1
239 
240  distmarkers(bottom:top) = nint((cliffheights(bottom:top) - distlow) / dzdist)
241  do j = bottom, top ! Remove possible rounding errors
242  if(distmarkers(j) < 1) distmarkers(j)=1
243  if(distmarkers(j) > npp-1) distmarkers(j)=npp-1
244  end do
245 
246  this_dist(bottom:top) = xshore(distmarkers(bottom:top))/ &
247  sum(xshore(distmarkers(bottom:top)))
248 
249  ! If have alloced larger than allocCeil reset to allocCeil (JT221107a)
250  if (half_tide > allocceil)then
251  deallocate(xshore, tide_duration, start_fnctn, end_fnctn, temp, stat=alloc_error)
252  if(alloc_error /= 0)then
253  write(6,*)'*** Unexpected deallocation error'
254  write(6,*)'*** for subroutine xshore_dist_t'
255  endif
256 
257  npp = max_np_sf + 2 * allocceil
258  allocate(xshore(npp), &
259  tide_duration(2 * allocceil + 1), temp(npp), &
260  start_fnctn(npp), end_fnctn(npp), &
261  stat = alloc_error)
262  success_code = success_code_from_allocation(alloc_error, 'subroutine xshore_dist_t end')
263  if (success_code < 0) return
264  endif
265  ! end of dealloc
266 
267  return
268  end subroutine xshore_dist_t

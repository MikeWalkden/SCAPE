158  jmax = 0
159  jmin = 0
160 
161  do j = 1, nce
162  if (sect_drift(j, section) > 0.0d0) jmax = j
163  if (jmin == 0 .and. sect_drift(j, section) > 0.0d0) jmin = j
164  end do
165 
166  if (jmin == 0 .or. jmax == jmin) then
167  ! No range of drift or only one beach elements within the range of drift.
168  ! Use the whole beach beach tangent, up to 3m beach height.
169  jmax = beachcrestlevele

171  end if

174  end do
175  end if

177  call exchange_material(nysections, offshore_st, bvolume, obvolume, h, period, tstep_secs)
178 
179  ! This code excludes waves from inappropriate directions
180 
181  do qpoint = 1, nqpoints
182  if ((theta_qpoint(qpoint) < 998.0d0) .and. &
183  ((excludewaves == 0) .or. ((theta_qpoint(qpoint) <= right_exclusion_angle(qpoint)) .and. &
184  (theta_qpoint(qpoint) >= -left_exclusion_angle(qpoint))))) then
185 
186  if (usealtsedimenttransport) then ! Use Alternate sediment transport calculation
187  if (qpoint == 1) then
188  tanbeachqpoint = tanbeachalt(1)
189  else if (qpoint == nqpoints) then
190  tanbeachqpoint = tanbeachalt(nysections)
191  else
192  tanbeachqpoint = (tanbeachalt(qpoint - 1) + tanbeachalt(qpoint)) * 0.5d0
193  end if
194 
195  ! Alternate longshore sediment transport equation.
196  qp(qpoint) = factor * (h_qpoint(qpoint)**3.1) * (tanbeachqpoint**0.4) * sin(2.0d0 * theta_qpoint(qpoint))
197 
198  else ! Use the CERC equation of longshore sediment transport.
199  qp(qpoint) = h_qpoint(qpoint) * h_qpoint(qpoint) * cgroup_qpoint(qpoint) * &
200  sin(2.0d0 * theta_qpoint(qpoint)) * cerck1(qpoint) / 37.8d0
201  end if
202  end if

206  ! update effect of groynes
207  !*************************************************
208 
209  if (includegroynes) then
210  ! Groynes on the Q sections
211  do qpoint = 1, nqpoints
212  ! Installation of groynes,
213 
214  ! ------------- MC: 101003, Groyne Bypassing
215 
216  if (year >= groyne_construction(qpoint) .and. (groyne_construction(qpoint) > 0) &
217  .and. year < groyne_removal(qpoint))then
218 
219  ! MC: 10 Oct 2003
220  ! For sections in which groynes have been built, this bit of code
221  ! checks what the beach width at that section is and adjusts the
222  ! sediment transport rate through the groyne effect value
223 
224  ! beach width was calculated on the Y sections
225  ! but that data needed at the Q sections
226 
227  ! Find updrift section
228  if (qp(qpoint) > 0.0d0) then
229  section = qpoint - 1
230  else
231  section = qpoint
232  end if
233 
234  if (section /= 0 .and. section /= nqpoints) then
235  !Updrift section is inside model boundary
236  beach_temp = 0.0d0
237  cliff_temp = 0.0d0
238  where (beachy(:, section) >= cliffy(:, section))
239  beach_temp = beachy(:, section)
240  cliff_temp = cliffy(:, section)
241  end where
242 
243  beachwidth = maxval(beach_temp) - minval(beach_temp, mask = beach_temp > 0.0d0)
244  bermwidth = minval(beach_temp, mask = beach_temp > 0.0d0) - minval(cliff_temp, mask = cliff_temp > 0.0d0)
245  beachwidth = beachwidth + bermwidth
246 
247  groyneeffect = groyne_effect_lookup(beachwidth)
248 
249  if (groyneeffect /= 0.0d0) then
250  qp(qpoint) = qp(qpoint) * (1.0d0 - groyneeffect)
251  groyneeffectannualsum(qpoint) = groyneeffectannualsum(qpoint) + groyneeffect
252  end if
253  end if
254  end if
255  end do ! qPoint = 1, nQpoints
256  end if
257 
258  ! Extrapolate the boundary conditions from the two sections just inside the model
259  !

261  qp(nqpoints) = qp(nqpoints-1) - ((qp(nqpoints-2)-qp(nqpoints-1)))
262  !
263 
264  qp = qp * tstep_secs
265 
266  if (qp(1) > 0.0d0) then
267  ! influx at right boundary
268  if (qp(1) > qpmaxboundaryrightin)then
269  qp(1) = qpmaxboundaryrightin
270  end if
271  else
272  ! outflux at right boundary
273  if (qp(1) < -qpmaxboundaryrightout)then
274  qp(1) = -qpmaxboundaryrightout
275  end if
276  end if
277 
278  if (qp(nqpoints) > 0.0d0) then
279  ! outflux at left boundary
280  if (qp(nqpoints) > qpmaxboundaryleftout)then
281  qp(nqpoints) = qpmaxboundaryleftout
282  end if
283  else
284  ! influx at left boundary
285  if (qp(nqpoints) < -qpmaxboundaryleftin)then
286  qp(nqpoints) = -qpmaxboundaryleftin
287  end if
288  end if

299

302 
303  do qpoint = 2, nqpoints - 1 ! Not able to deal with the boundaries here
304  if(qp(qpoint) < 0)then
305  q(qpoint) = qp(qpoint) * drift_fraction(qpoint)
306  else
307  q(qpoint) = qp(qpoint) * drift_fraction(qpoint - 1)
308  endif
309  end do

312 ! The boundaries are constricted by the drift fraction just inside the model

328  end if

337  end do

340 ! The sediment balancing algorithms have been checked out on page 40 WkBk 6
341  do while(volcheck < 0) ! if this is less than zero

343  if(bvolnew(section) < 0)then ! It is section profile

351  else ! Only flow away downchainage

353  end if
354  else ! only flow away upchainage
355  q(section+1) = floor(bvolume(section) + q(section))
356  end if
357  exit ! so that it only corrects one beach volume at a time
358  end if
359  end do
360  do section = 1, nysections
361  bvolnew(section) = nint(bvolume(section) + q(section) - q(section + 1))
362  end do
363  volcheck = minval(bvolnew)
364  end do
365

367  ! Looking offshore, left end is at Q(nQpoints), right is at Q(1) and +ve Q is right to left
368  pot_left_sediment_flux = qp(nqpoints)
369  left_sediment_flux = q(nqpoints)
370 
371  pot_right_sediment_flux = qp(1)
372  right_sediment_flux = q(1)
373 
374  ! Finish updating the beach volumes
375  bvolume = bvolnew
376 
377  ! Inject sediment_influx (if any) from OpenMI
378  bvolume = bvolume + sediment_influx
379 
380  ! code to record the average sediment transport each year
381  qann = qann + q
382  qpann = qpann + qp
383

386  deallocate(tanbeachalt, stat = alloc_error)
387  end if

403  h, period, tstep_secs)

407  implicit none

409  integer, intent(in) :: nYsections !< number of beach sections
410  real(kind=double), intent(out), dimension(nYsections) :: offshore_st !< Offshore sediment transport
411  real(kind=double), intent(inout), dimension(nYsections) :: bvolume !< beach volume
412  real(kind=double), intent(inout), dimension(nYsections) :: obvolume !< Offshore bar volume
413  real(kind=double), intent(in), dimension(nYsections) :: h !< Wave height
414  real(kind=double), intent(in) :: period !< Wave period
415  real(kind=double), intent(in) :: tstep_secs !< Time step in seconds
416 
417  real(kind=double) :: offstrLookup
418  integer :: section
419 
420  offshore_st = 0
421 
422  do section = 1, nysections
423  if (h(section) > minstormwaveheight) then
424  offstrlookup = offshore_str_lookup(h(section), period)
425  offshore_st(section) = dx * offstrlookup * tstep_secs / 3600 ! convert
426  ! to units suitable for this timestep
427  if(-1*offshore_st(section) > bvolume(section))then
428  offshore_st(section) = -1*bvolume(section)
429  endif
430  else ! Not a storm condition so the material moves onshore
431  offshore_st(section) = obvolume(section) * beachreturnratio
432  endif
433 
434  bvolume(section) = bvolume(section) + offshore_st(section)
435  obvolume(section) = obvolume(section) - offshore_st(section)
436 
437  if (obvolume(section) < 1.0d-6) obvolume(section) = 0.0d0
438 
439  enddo
